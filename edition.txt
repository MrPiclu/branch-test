#define SIZE 3;

};

typedef struct Lotto lt;

void lottoList(Lotto *lt);
void setName(Lotto *lt);
void setAge(Lotto *lt);

int main(){
	srand(time(NULL));
	Lotto lt[3];
	for(int i = 0; i < SIZE; i++){
		setName(&lt[i]);
		setAge(&lt[i]);
		lottoList(&lt[i]);
	}
	for (int i = 0; i < SIZE; i++){
		printf("%s %d", lt[i].name, lt[i].age);
		for ( int k = 0; k < 6; k++){
			printf(" %d", lt[i].list[k]);
	
		}
		printf("\n");
	}
	return 0;
}
